at the end of each line etc set -x # show commands set -e # stop on error # assume we are running on ubuntu distribname='other'; if [[ -f /etc/lsb-release ]]; then { distribname=$(cat /etc/lsb-release |grep DISTRIB_CODENAME | cut -d= -f 2); } fi; springheadlessurl="http://hughperkins.s3.amazonaws.com/springheadless_${distribname}.tar.bz2"; springgridgitclonepath="git://github.com/hughperkins/springgrid.git"; export JAVA_HOME="/usr/lib/jvm/java-6-openjdk"; boostversion=40; if [[ ${distribname} == jaunty ]]; then { boostversion=37; } fi; sudo apt-get update; sudo dpkg --configure -a # in case we interrupted the script and rerun it sudo apt-get --yes install git-core; sudo apt-get --yes install openjdk-6-jre-headless; sudo apt-get --yes install libopenal1 libogg0 libvorbisfile3 libfreetype6 libxcursor1 libboost-regex1.${boostversion}.0 libboost-thread1.${boostversion}.0 libboost-program-options1.${boostversion}.0 libboost-system1.${boostversion}.0 libboost-signals1.${boostversion}.0 libglew1.5 libdevil1c2; echo "Downloading prerequisites for building AIs..."; sudo apt-get --yes install build-essential p7zip-full; if [[ ${distribname} == jaunty ]]; then { sudo apt-get --yes install cmake; } fi; if [[ ${distribname} == karmic ]]; then { sudo apt-get --yes install cmake-curses-gui; } fi; sudo apt-get --yes install build-essential p7zip-full; sudo apt-get --yes install zlib1g-dev libfreetype6-dev libsdl1.2-dev libopenal-dev libglew-dev zip libvorbis-dev libxcursor-dev libdevil-dev libboost-regex1.${boostversion}-dev libboost-thread1.${boostversion}-dev libboost-program-options1.${boostversion}-dev libboost-system1.${boostversion}-dev libboost-signals1.${boostversion}-dev; cd; if [[ -d $HOME/springgrid ]]; then { # means we can run this script multiple times # without errors rm -Rf $HOME/springgrid } fi; git clone $springgridgitclonepath; mkdir -p $HOME/git; cd $HOME/git; if [[ ! -f $(basename $springheadlessurl) ]]; then { # save downloading it multiple # times echo "downloading springheadless.tar.bz2 ..."; wget $springheadlessurl; } fi; echo "unpacking springheadless.tar.bz2 ..."; tar -xjf $(basename $springheadlessurl); cd $HOME/springgrid/botrunner; export SPRING_DATADIR=$HOME/git/springheadless/game; SPRINGSRCDIR=$HOME/git/springheadless python botrunner.py --target-web-site={{websiteurl}} --botrunner-name=cloud-$RANDOM --botrunner-shared-secret=foo --spring-source-path=$SPRINGSRCDIR --spring-path=$SPRINGSRCDIR/game/spring-hl --unitsync-path=$SPRINGSRCDIR/game/libunitsync.so --downloading-ok --yes;
